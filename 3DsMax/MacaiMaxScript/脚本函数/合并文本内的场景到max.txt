-- merge objects named in a name text file from given scene
fn merge_by_namefile scenefile namefile dups:#mergeDups makeUnique:true =
(
	local f = openFile namefile,
		  names = #()
	while not eof f do append names (readValue f)
	close f
	mergeMAXFile scenefile names #select dups
	if makeUnique then make_selnames_unique ()
)
