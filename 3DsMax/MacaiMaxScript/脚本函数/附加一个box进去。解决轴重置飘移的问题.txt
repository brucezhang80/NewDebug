	--附加一个box进去。解决轴重置飘移的问题
	fn AttachToAbox =
	(
		max select all
		myselection = selection
		if myselection.count>0 then (
		curname = myselection[1].name
		mybox = box()
		mymesh = converttomesh mybox
			
		for m in myselection do (
			meshop.attach mymesh m
		)

		mymesh.name = curname

		clearselection()

			for i = 1 to 12 do (
				deleteface mymesh 1
			)
		)
	)
